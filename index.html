    2. npm install jquery --save
    3. npm install angular --save
    4. npm install bootstrap --save
  • Create app.js
  • Create index.html
    1. emmet !
    2. in the html tag add ng-app attribute
    3. in head, create a with href attribute pointing to ./node_modules/bootstrap/dist/css/bootstrap.css
    4. in the body tag add ng-controller attribute with name mainCtrl
    5. in body, create a div with inner text containing angular syntax such as: Angular is {{2>1}}ly working!
    6. emmet script:src with src attribute pointing to ./node_modules/jquery/dist/jquery.min.js
    7. emmet script:src with src attribute pointing to ./node_modules/bootstrap/dist/js/bootstrap.min.js
    8. emmet script:src with src attribute pointing to ./node_modules/angular/angular.min.js
    9. emmet script:src with src attribute pointing to app.js

    1. var app = angular.module('', []);
